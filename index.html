# Install VEIN
install.packages("vein")
library(vein)

# Estimate emissions
emisions <- emis(
  veh = "PC", 
  size = "<1400cc", 
  fuel = "G", 
  pollutant = "CO",
  speed = speed_data
)

plot(emisions, col="#00FF88")
